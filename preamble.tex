			\ifthenelse{\equal{#1}{I}}{\item[\textbf{Indexes}]}{%
				\ifthenelse{\equal{#1}{A}}{\item[\textbf{Abbreviations}]}{%
					\ifthenelse{\equal{#1}{O}}{\item[\textbf{Operators}]}{}

			}
		}
	}
